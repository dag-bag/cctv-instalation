# Next.js + Cloudflare (The Final Boss Config)
# SEO & Performance Optimized
# ==========================================

# 1. HTTP to HTTPS Redirect (Backup)
server {
    listen 80;
    listen [::]:80;
    server_name camharbor.in www.camharbor.in;
    
    # Cloudflare normally handles this, but 301 is best for SEO safety
    return 301 https://camharbor.in$request_uri;
}

# 2. WWW to Non-WWW Redirect (Crucial for SEO)
# Google considers www and non-www as different sites. This fixes it.
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name www.camharbor.in;

    ssl_certificate /etc/letsencrypt/live/camharbor.in/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/camharbor.in/privkey.pem;

    return 301 https://camharbor.in$request_uri;
}

# 3. MAIN SERVER (HTTPS - Non-WWW)
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name camharbor.in;

    # SSL Certificates (Let's Encrypt)
    ssl_certificate /etc/letsencrypt/live/camharbor.in/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/camharbor.in/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Gzip Compression
    gzip on;
    gzip_min_length 256;
    gzip_types text/plain text/css application/json application/javascript image/svg+xml;

    # Security Headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;

    # Static Assets Caching (Cloudflare will honor this)
    location ~* \.(?:ico|css|js|gif|jpe?g|png|woff2?|eot|ttf|svg|webp)$ {
        proxy_pass http://127.0.0.1:3000;
        expires 30d;
        access_log off;
        add_header Cache-Control "public, no-transform";
    }

    # Main Next.js Proxy
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;

        # Standard Headers
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # WebSocket Support (Next.js HMR)
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # Buffer Tuning (Fast TTFB)
        proxy_buffering on;
        proxy_buffer_size 8k;
        proxy_buffers 16 8k;
        proxy_busy_buffers_size 16k;

        # Timeouts
        proxy_connect_timeout 60s;
        proxy_read_timeout 180s;
        proxy_send_timeout 180s;
    }

    # Logs
    access_log /var/log/nginx/camharbor.access.log;
    error_log  /var/log/nginx/camharbor.error.log;
}
